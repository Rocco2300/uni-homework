    ansi_c_string_to_date_time (s: READABLE_STRING_GENERAL): detachable DATE_TIME
            -- ANSI C date from s string representation using the RFC 1123
            --       asctime-date = wkday SP date3 SP time SP 4DIGIT
            --       date3        = month SP ( 2DIGIT | ( SP 1DIGIT ))
            --                      ; month day (e.g., Jun  2)
            --       time         = 2DIGIT ":" 2DIGIT ":" 2DIGIT
            --                      ; 00:00:00 - 23:59:59
            --       wkday        = "Mon" | "Tue" | "Wed"
            --                    | "Thu" | "Fri" | "Sat" | "Sun"
            --       month        = "Jan" | "Feb" | "Mar" | "Apr"
            --                    | "May" | "Jun" | "Jul" | "Aug"
            --                    | "Sep" | "Oct" | "Nov" | "Dec"			
        note
            eis: "name=RFC2616", "protocol=URI", "src=http://www.w3.org/Protocols/rfc2616/rfc2616-sec3.html"
        local
            t: STRING_32
            l_ddd, l_mmm: detachable STRING_32
            l_dd, l_yyyy, l_hh, l_mi, l_ss, l_ff2: INTEGER_32
            l_mo: INTEGER_32
            l_gmt_offset: INTEGER_32
            i, n: INTEGER_32
            err: BOOLEAN

            n := s.count

                i > n or else not s [i].is_space
            loop
                i := i + 1
            end;

            from

                i > n or else not s [i].is_alpha
            loop
                t.extend (s [i])
                i := i + 1
            end
            if i <= n and t.count >= 3 then
                l_ddd := t.as_lower
            else
                err := True
            end
            from
            until
                i > n or else not s [i].is_space
            loop
                i := i + 1
            end
            if not err then

                from

                    i > n or else not s [i].is_alpha
                loop
                    t.extend (s [i])
                    i := i + 1
                end
                if i <= n and t.count >= 3 then
                    l_mmm := t.as_upper
                    if l_mmm.count > 3 then
                        l_mmm.keep_head (3)
                    end
                    if l_mmm.same_string_general ("JAN") then
                        l_mo := 1
                    elseif l_mmm.same_string_general ("FEB") then
                        l_mo := 2
                    elseif l_mmm.same_string_general ("MAR") then
                        l_mo := 3
                    elseif l_mmm.same_string_general ("APR") then
                        l_mo := 4
                    elseif l_mmm.same_string_general ("MAY") then
                        l_mo := 5
                    elseif l_mmm.same_string_general ("JUN") then
                        l_mo := 6
                    elseif l_mmm.same_string_general ("JUL") then
                        l_mo := 7
                    elseif l_mmm.same_string_general ("AUG") then
                        l_mo := 8
                    elseif l_mmm.same_string_general ("SEP") then
                        l_mo := 9
                    elseif l_mmm.same_string_general ("OCT") then
                        l_mo := 10
                    elseif l_mmm.same_string_general ("NOV") then
                        l_mo := 11
                    elseif l_mmm.same_string_general ("DEC") then
                        l_mo := 12
                    else
                        err := True
                    end
                else
                    err := True
                end
            end
            from
            until
                i > n or else not s [i].is_space
            loop
                i := i + 1
            end
            if not err and i <= n then
                t.wipe_out
                from
                until
                    i > n or else not s [i].is_digit
                loop
                    t.extend (s [i])
                    i := i + 1
                end
                if i <= n and t.count > 0 then
                    check
                        t.is_integer
                    end
                    l_dd := t.to_integer
                else
                    err := True
                end
            else
                err := True
            end
            from
            until
                i > n or else not s [i].is_space
            loop

            end

                t.wipe_out

                until
                    i > n or else not s [i].is_digit

                    t.extend (s [i])
                    i := i + 1
                end
                if i <= n and t.count > 0 and s [i] = ':'.to_character_32 then
                    check
                        t.count = 2 and t.is_integer
                    end
                    l_hh := t.to_integer
                    i := i + 1
                else
                    err := True
                end
            end
            if not err then
                t.wipe_out
                from
                until
                    i > n or else not s [i].is_digit
                loop
                    t.extend (s [i])
                    i := i + 1
                end
                if i <= n and t.count > 0 and s [i] = ':'.to_character_32 then

                i := i + 1;
                t.wipe_out

                loop
                    t.extend (s [i])

                i := i + 1
            end

                from

                    i > n or else not s [i].is_digit
                loop
                    t.extend (s [i])
                    i := i + 1

                        t.count = 4 or t.count = 2 and t.is_integer

                    l_yyyy := t.to_integer
                    if t.count = 2 then

                    err := True

                i > n or else not s [i].is_space
            loop
                i := i + 1
            end
            if i <= n and not err then
                from
                until
                    i > n or else not s [i].is_space
                loop
                    i := i + 1
                end;
                t.wipe_out

                        i > n or else not s [i].is_space

                        i := i + 1

                            t.extend (s [i])
                        elseif s [i] = '-'.to_character_32 then
                            t.extend (s [i])
                        else
                            err := True
                        end
                        if not err then
                            i := i + 1
                            from
                            until
                                i > n or else not s [i].is_space
                            loop
                                i := i + 1
                            end
                            from
                            until
                                i > n or else not s [i].is_digit
                            loop
                                t.extend (s [i].as_upper)
                                i := i + 1
                            end
                            l_gmt_offset := t.to_integer * 60
                            if i <= n and s [i] = ':'.to_character_32 then
                                i := i + 1;
                                t.wipe_out
                                from
                                until
                                    i > n or else not s [i].is_digit
                                loop
                                    t.extend (s [i].as_upper)
                                    i := i + 1
                                end
                                l_gmt_offset := l_gmt_offset + l_gmt_offset.sign * t.to_integer
                            end
                        end
                    end
                else
                    err := True
                end
            end
            if not err then
                check
                    valid_yyyy: 0 < l_yyyy
                    valid_dd: 0 < l_dd and l_dd <= 31
                    valid_mo: 0 < l_mo and l_mo <= 12
                    valid_hh: 0 <= l_hh and l_hh <= 23
                    valid_mi: 0 <= l_mi and l_mi <= 59

                if l_gmt_offset /= 0 then
                    Result.minute_add (- l_gmt_offset)
                end
            end

    rfc3339_string_to_date_time (s: READABLE_STRING_GENERAL): detachable DATE_TIME
        note
            eis: "name=RFC3339", "protocol=URI", "src=https://tools.ietf.org/html/rfc3339"
        local
            y, mo, d, h, mi, sec, frac_sec: INTEGER_32
            l_off_h, l_off_mi: INTEGER_32
            tmp: READABLE_STRING_GENERAL
            i, j, k, pos: INTEGER_32
            err: BOOLEAN
        do
            if s /= Void then
                pos := s.index_of ('T'.to_character_32, 1)
                if pos > 0 then
                    i := 1
                    j := s.index_of ('-'.to_character_32, i)
                    if j > i and j < pos then
                        y := s.substring (i, j - 1).to_integer
                        i := j + 1
                        j := s.index_of ('-'.to_character_32, i)
                        if j > i and j < pos then
                            mo := s.substring (i, j - 1).to_integer
                            d := s.substring (j + 1, pos - 1).to_integer
                        else
                            err := True
                        end
                    else
                        err := True
                    end
                    i := pos + 1
                    j := s.index_of (':'.to_character_32, i)
                    if j > i then
                        h := s.substring (i, j - 1).to_integer
                        i := j + 1
                        j := s.index_of (':'.to_character_32, i)
                        if j > i then
                            mi := s.substring (i, j - 1).to_integer
                            from
                                k := j + 1
                            until
                                k > s.count or s [k].is_alpha or s [k] = '+'.to_character_32 or s [k] = '-'.to_character_32
                            loop
                                k := k + 1
                            end
                            tmp := s.substring (j + 1, k - 1)
                            j := tmp.index_of ('.'.to_character_32, 1)
                            if j > 0 then
                                sec := tmp.head (j - 1).to_integer
                                frac_sec := tmp.substring (j + 1, tmp.count).to_integer
                            else
                                sec := tmp.to_integer
                            end
                            if k <= s.count then
                                if s [k] = '+'.to_character_32 then
                                    tmp := s.substring (k + 1, s.count)
                                    j := tmp.index_of (':'.to_character_32, 1)
                                    if j > 0 then
                                        l_off_h := - tmp.head (j - 1).to_integer
                                        l_off_mi := - tmp.substring (j + 1, tmp.count).to_integer
                                    else
                                        l_off_h := - tmp.to_integer
                                    end
                                elseif s [k] = '-'.to_character_32 then
                                    tmp := s.substring (k + 1, s.count)
                                    j := tmp.index_of (':'.to_character_32, 1)
                                    if j > 0 then
                                        l_off_h := + tmp.head (j - 1).to_integer
                                        l_off_mi := + tmp.substring (j + 1, tmp.count).to_integer
                                    else
                                        l_off_h := + tmp.to_integer
                                    end
                                elseif s [k] = 'Z'.to_character_32 then
                                end
                            end
                        else
                            err := True
                        end
                    else
                        err := True
                    end
                    if not err then
                        create Result.make (y, mo, d, h, mi, sec)
                        if frac_sec /= 0 then
                            Result.set_fractionals (frac_sec.to_double);
                            Result.time.set_fractionals (frac_sec.to_double)
                        end
                        if l_off_h /= 0 then
                            Result.hour_add (l_off_h)
                        end
                        if l_off_mi /= 0 then
                            Result.minute_add (l_off_mi)
                        end
                    end
                end
            end
        end
