                    before := False
                    counter := 1

                until

                    p := p.right
                    counter := counter + 1

                    after := True
                else
                    active := p
                end

                new_index := index + i
                before := True
                after := False

                if new_index > 0 then
                    move (new_index)
                end
            end
        ensure then

            after_set: (old index + i) >= (count + 1) implies after
        end
