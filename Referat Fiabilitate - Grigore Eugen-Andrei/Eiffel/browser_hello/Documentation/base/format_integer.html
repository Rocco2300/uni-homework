    process_sign (s: STRING_8; sn: INTEGER_32): STRING_8
            -- Process sign related values.
        local
            sstring: detachable STRING_8
        do
            Result := s
            if bracketted_negative and sn = -1 then
                Result.precede ('(');
                Result.extend (')')

            if ((show_sign_negative or show_sign) and sn = -1) or else ((show_sign_positive or show_sign) and sn = 1) or else (show_sign and sn = 0) then

            end
            if sstring /= Void then

    justify (s: STRING_8): STRING_8

            room: s.count < width
            justification: justification /= No_justification
        local
            l, r: STRING_8
            i, t: INTEGER_32
        do
            Result := s
            if not centered then
                create l.make (width - s.count)
                from
                    i := 1
                until

                loop
                    l.extend (fill_character)
                    i := i + 1
                end
                if left_justified then
                    Result.append (l)
                else
                    Result.prepend (l)
                end
            else
                t := (width - s.count) // 2
                if 2 * t + s.count < width then
                    create l.make (t + 1)
                else
                    create l.make (t)
                end
                create r.make (t)
                from
                    i := 1
                until
                    i > r.capacity
                loop
                    l.extend (fill_character);
                    r.extend (fill_character)
                    i := i + 1
                end
                if i = l.capacity then
                    l.extend (fill_character)
                end;
                Result.prepend (l);
                Result.append (r)
            end
        end
